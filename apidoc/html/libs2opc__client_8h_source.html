283 /*
284  @description
285  Connection configuration to a remote OPC server
286  @field server_url
287  Zero-terminated path to server URL
288  @field security_policy
289  The chosen OPC-UA security policy for the connection, one of the SOPC_SecurityPolicy_*_URI string
290  @field security_mode
291  The chosen OPC-UA security mode for the connection, one of the OpcUa_MessageSecurityMode constant
292  @field disable_certificate_verification
293  Uses a PKIProvider which does not verify the certificates against a certificate authority.
294  Setting this flag to not 0 severely harms security. The certificate authority is not required in this case.
295  Other certificates are still required when using an encrypted or signed secure channel.
296  @field path_cert_auth
297  Zero-terminated path to the root certificate authority in the DER format
298  @field path_cert_srv
299  Zero-terminated path to the server certificate in the DER format, signed by the root certificate authority
300  @field path_cert_cli
301  Zero-terminated path to the client certificate in the DER format, signed by the root certificate authority
302  @field path_key_cli
303  Zero-terminated path to the client private key which is paired to the public key signed server certificate,
304  in the DER format
305  @field path_crl
306  Zero-terminated path to the certificate revocation list in the DER format
307  @field policyId
308  Zero-terminated policy id. To know which policy id to use, please read a
309  GetEndpointsResponse or a CreateSessionResponse. When username is NULL, the
310  AnonymousIdentityToken is used and the policy id must correspond to an
311  anonymous UserIdentityPolicy. Otherwise, the UserNameIdentityToken is used
312  and the policy id must correspond to an username UserIdentityPolicy.
313  @field username
314  Zero-terminated username, NULL for anonymous access, see policyId
315  @field password
316  Zero-terminated password, ignored when username is NULL. Password is kept in memory for future reconnections.
317  @field publish_period_ms
318  The requested publish period for the created subscription (in milliseconds)
319  @field n_max_keepalive
320  The max keep alive count for the subscription. When there is no notification to send, the subscription waits
321  at most the number of publish cycle before sending a publish response (which is then empty).
322  @field n_max_lifetime
323  The max number of time that a subscription may timeout its publish cycle without being able to send a
324  response (because there is no publish request to answer to). In this case, the subscription is killed by the
325  server. A large value is recommended (e.g. 1000).
326  @field data_change_callback
327  The callback for data change notification
328  @field timeout_ms
329  Connection timeout (milliseconds)
330  @field sc_lifetime
331  Time before secure channel renewal (milliseconds). A parameter larger than 60000 is recommended.
332  @field token_target
333  Number of tokens (PublishRequest) that the client tries to maintain throughout the connection
334  @field generic_response_callback
335  The callback used to transmit generic responses to request passed
336  through SOPC_LibSub_AsyncSendRequestOnSession.
337  */
338 typedef struct
339 {
340  SOPC_LibSub_CstString server_url;
341  SOPC_LibSub_CstString security_policy;
342  OpcUa_MessageSecurityMode security_mode;
343  uint8_t disable_certificate_verification;
344  SOPC_LibSub_CstString path_cert_auth;
345  SOPC_LibSub_CstString path_crl;
346  SOPC_LibSub_CstString path_cert_srv;
347  SOPC_LibSub_CstString path_cert_cli;
348  SOPC_LibSub_CstString path_key_cli;
349  SOPC_LibSub_CstString policyId;
350  SOPC_LibSub_CstString username;
351  SOPC_LibSub_CstString password;
352  int64_t publish_period_ms;
353  uint32_t n_max_keepalive;
354  uint32_t n_max_lifetime;
355  SOPC_LibSub_DataChangeCbk data_change_callback;
356  int64_t timeout_ms;
357  uint32_t sc_lifetime;
358  uint16_t token_target;
359  SOPC_LibSub_EventCbk generic_response_callback;
360 } SOPC_LibSub_ConnectionCfg;
