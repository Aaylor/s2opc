65 /*
66  @description
67  Connection configuration to a remote OPC server
68  @field security_policy
69  The chosen OPC-UA security policy for the connection, one of the SOPC_SecurityPolicy_*_URI string.
70  zero-terminated string
71  @field security_mode
72  The chosen OPC-UA security mode for the connection.
73  The list of accepted values is:
74  - 1: no security mode,
75  - 2: only signature,
76  - 3: signature and encryption.
77  @field path_cert_auth
78  Zero-terminated path to the root certificate authority in the DER format
79  @field path_crl
80  Zero-terminated path to the Certificate Revocation List (CRL) of the certificate authority in the DER format
81  @field path_cert_srv
82  Zero-terminated path to the server certificate in the DER format, signed by the root certificate authority
83  @field path_cert_cli
84  Zero-terminated path to the client certificate in the DER format, signed by the root certificate authority
85  @field path_key_cli
86  Zero-terminated path to the client private key which is paired to the public key signed client certificate,
87  in the DER format
88  @field policyId
89  Zero-terminated policy id. To know which policy id to use, please read a
90  GetEndpointsResponse or a CreateSessionResponse. When username is NULL, the
91  AnonymousIdentityToken is used and the policy id must correspond to an
92  anonymous UserIdentityPolicy. Otherwise, the UserNameIdentityToken is used
93  and the policy id must correspond to an username UserIdentityPolicy.
94  @field username
95  Zero-terminated username, NULL for anonymous access, see policyId
96  @field password
97  Zero-terminated password, ignored when username is NULL. Password is kept in memory for future reconnections.
98 */

100 {
101  const char* security_policy;
102  int32_t security_mode;
103  const char* path_cert_auth;
104  const char* path_crl;
105  const char* path_cert_srv;
106  const char* path_cert_cli;
107  const char* path_key_cli;
108  const char* policyId;
109  const char* username;
110  const char* password;
111 } SOPC_ClientHelper_Security;
